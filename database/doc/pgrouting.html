Function: closest_edge(y numeric, x numeric)

Returns: SET OF edge

Language: PLPGSQL

DECLARE
  --range NUMERIC (12, 8) DEFAULT 0.002  ; -- ~222 m at equator
  range NUMERIC (12, 8) DEFAULT 0.2; -- ~22.2 km at equator
  srid  INTEGER         DEFAULT 4326;  -- WGS84
  bbox  VARCHAR(256);
  query TEXT;
  result record;
  edge edge;
BEGIN
  bbox := 'BOX3D(' || x - range || ' ' || y - range || ',' || x + range || ' ' || y + range || ')';

  query := 'SELECT gid AS id,
                   source AS startnode,
                   target AS endnode
              FROM network.edges
             WHERE the_geom && ST_SetSRID (''' || bbox || '''::box3d, ' || srid || ')
          ORDER BY distance(the_geom, GeometryFromText(''POINT(' || x || ' ' || y || ')'', ' || srid || '))
             LIMIT 1';
	FOR result IN EXECUTE query
    LOOP
      edge.id        := result.id;
      edge.startnode := result.startnode;
      edge.endnode   := result.endnode;
      RETURN NEXT edge;
    END LOOP;
END;

Function: closest_edge_node(range numeric, y numeric, x numeric)

Returns: SET OF edge_node

Language: PLPGSQL

DECLARE
  srid       INTEGER         DEFAULT 4326;  -- WGS84
  bbox       VARCHAR(256);
  query      TEXT;
  result     record;
  edge_node  edge_node;
BEGIN
  bbox := 'BOX3D(' || x - range || ' ' || y - range || ',' || x + range || ' ' || y + range || ')';

  query := 'SELECT e.gid AS edge_id,
                   CASE WHEN 
                          distance (ep.geom, ST_StartPoint (e.the_geom) ) >= distance (ep.geom, ST_EndPoint (e.the_geom)) 
                        THEN
                          e.source
                        ELSE
                          e.target
                   END AS nodeid
              FROM network.edges AS e
              JOIN network.edge_points AS ep
                   ON e.gid = ep.edge_id 
             WHERE ep.geom && ST_SetSRID (''' || bbox || '''::box3d, ' || srid || ')
          ORDER BY distance(ep.geom, GeometryFromText(''POINT(' || x || ' ' || y || ')'', ' || srid || '))
             LIMIT 1';

  FOR result IN EXECUTE query
  LOOP
    edge_node.edge_id := result.edge_id;
    edge_node.node_id := result.node_id;

    RETURN NEXT edge_node;
  END LOOP;
  RETURN;
END

DECLARE

  bbox := 'BOX3D(' || x - range || ' ' || y - range || ',' || x + range || ' ' || y + range || ')';

              FROM network.edge_points

          ORDER BY distance(geom, GeometryFromText(''POINT(' || x || ' ' || y || ')'', ' || srid || '))
             LIMIT 1';

  LOOP
      edge_point.edge_id := result.edge_id;
      edge_point.point   := result.point;
      edge_point.x       := result.x;
      edge_point.y       := result.y;
      edge_point.error   := result.error;
      RETURN NEXT edge_point;
  END LOOP;
  RETURN;
END

Function: closest_nodes(number numeric, range numeric, y numeric, x smallint)

Returns: SET OF node

Language: PLPGSQL

DECLARE
  srid   INTEGER       DEFAULT 4326;  -- WGS84
  bbox   VARCHAR(256);
  query  TEXT;
  result record;
  node   network.node;
BEGIN
  bbox := 'BOX3D(' || x - range || ' ' || y - range || ',' || x + range || ' ' || y + range || ')';

  query := 'SELECT id,
                   ST_X(the_geom) AS x,
                   ST_Y(the_geom) AS y,
                   distance(the_geom, GeometryFromText(''POINT(' || x || ' ' || y || ')'', ' || srid || ')) AS error
              FROM network.nodes
             WHERE the_geom && ST_SetSRID (''' || bbox || '''::box3d, ' || srid || ')
          ORDER BY distance(the_geom, GeometryFromText(''POINT(' || x || ' ' || y || ')'', ' || srid || '))
             LIMIT ' || number;

	FOR result IN EXECUTE query
    LOOP
      node.id    := result.id;
      node.x     := result.x;
      node.y     := result.y;
      node.error := result.error;
      RETURN NEXT node;
    END LOOP;
END;
